enviarTexto(texto: string): void {

onResultadoTexto(): Observable<{ resultado: string }> {
  return new Observable(observer => {
    const handler = (data: any) => observer.next(data);
    this.socket.on('resultado-palabras', handler);
    return () => this.socket.off('resultado-palabras', handler);
  });
}




server.js — añadir dentro del bloque io.on('connection', socket)


socket.on('contar-palabras', ({ texto }) => {
  const palabras = texto.trim().split(/\s+/).length;
  const resultado = `${texto.toUpperCase()} — ${palabras} palabras`;
  socket.emit('resultado-palabras', { resultado });
});
